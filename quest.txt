SELECT 
    CTJD.TRADEJURISDICTIONDETAILSBUYERSELLERMIFID2ID,
    CTJDM.TRADEJURISDICTIONDETAILSMIFID2ID,
    CTJ.TRADEJURISDICTIONID,
    CT.TRADEID,
    CT.CTYSDSID,
    CT.BOOK
FROM 
    CASH_TRADEJURISDICTIONDETAILSBUYERSELLERMIFID2 CTJD
JOIN 
    CASH_TRADEJURISDICTIONDETAILSMIFID2 CTJDM
    ON CTJD.TRADEJURISDICTIONDETAILSMIFID2ID = CTJDM.TRADEJURISDICTIONDETAILSMIFID2ID
JOIN 
    CASH_TRADEJURISDICTION CTJ
    ON CTJDM.TRADEJURISDICTIONID = CTJ.TRADEJURISDICTIONID
JOIN 
    CASH_TRADE CT
    ON CTJ.TRADEID = CT.TRADEID
WHERE 
    CT.brr_load_dt BETWEEN '2023-01-01' AND '2023-03-31'
    AND CTJD.onpremflag = 1
    AND CTJDM.onpremflag = 1
    AND CTJ.onpremflag = 1
    AND CT.onpremflag = 1;
